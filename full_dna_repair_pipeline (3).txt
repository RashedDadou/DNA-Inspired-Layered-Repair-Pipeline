def full_dna_repair_pipeline(
    self,
    input_image: Image.Image,
    raw_prompt: str,
    mask: Optional[Image.Image] = None,
    reference_face: Optional[Image.Image] = None,  # لـ IP-Adapter FaceID
    use_latent_pulse: bool = False,
    use_ip_adapter: bool = True,
    pulse_steps: int = 6,
    positive_boost: float = 0.28,
    negative_suppress: float = 0.22,
    latent_strength_base: float = 0.18,
    latent_steps_per_pulse: int = 12,
    use_colored_layers: bool = True,
    blend_mode: Literal["density", "wave"] = "density",
    final_sharpen: float = 1.14,
    final_contrast: float = 1.06,
    final_color: float = 1.04,
    max_retries: int = 2,
    output_path: str = "dna_repaired_final.jpg",
) -> Tuple[Image.Image, Dict]:
    """
    الـ Pipeline النهائي الكامل للترميم الـ DNA-inspired
    يجمع كل الطبقات في سير عمل واحد متكامل
    """
    # ── 1. Input Prompt / Scene + Filter Prompt + Scene DNA Genes ────────
    filtered_prompt, scene_genes = prepare_scene_prompt_and_genes(
        raw_prompt=raw_prompt,
        image=input_image,
    )
    print("Filtered Prompt:", filtered_prompt)
    print("Scene DNA Genes:", json.dumps(scene_genes, indent=2, ensure_ascii=False))

    # إنشاء ماسك إذا ما كان موجود
    if mask is None:
        mask = detect_dead_zones(input_image)  # من الطبقة السابقة

    # ── 2. ControlNet Structure (إنتاج الشبكة Net) ────────────────────────
    net = self.generate_net_structure(
        img=input_image,
        mask=mask,
        control_type="union",
        net_strength=0.68,
        steps=18,
    )

    # ── 3. DNA Light Layer (الشفافة الأولى) ──────────────────────────────
    dna_layer1 = self.create_dna_light_layer(input_image.size, opacity=0.38)

    # ── 4. Geometric Repair via Net (مع IP-Adapter إذا موجود) ─────────────
    repaired_geometry = self.geometric_repair_via_net(
        img=input_image,
        mask=mask,
        net=net,
        prompt=filtered_prompt,
        strength=0.32,
        use_ip_adapter=use_ip_adapter,
        reference_face=reference_face,
    )

    # ── 5. Colored DNA Layers (طبقات لونية مخصصة) ─────────────────────────
    if use_colored_layers:
        colored_layers = self.add_colored_dna_layers(
            net=net,
            mask=mask,
            blend_mode=blend_mode,
            opacity=0.52,
        )
        repaired_geometry = Image.alpha_composite(repaired_geometry.convert("RGBA"), colored_layers)

    # ── 6. DNA-inspired Color Pulse (pixel أو latent) ───────────────────────
    pulsed_image = self.dna_inspired_color_pulse(
        img=repaired_geometry,
        mask=mask,
        prompt=filtered_prompt,
        pulse_steps=pulse_steps,
        positive_boost=positive_boost,
        negative_suppress=negative_suppress,
        latent_mode=use_latent_pulse,
        pipeline=self.pipeline,
        latent_strength_base=latent_strength_base,
        latent_steps_per_pulse=latent_steps_per_pulse,
    )

    # ── 7. Monitoring Module (المحسن) ──────────────────────────────────────
    monitoring_report = self.monitor_repair_result(
        original=input_image,
        repaired=pulsed_image,
        mask=mask,
    )
    print("\nMonitoring Report:")
    print(json.dumps(monitoring_report, indent=2, ensure_ascii=False))

    # ── 8. Post-Monitor Filter + Decision Flow ──────────────────────────────
    correction_prompt, needs_retry = self.post_monitor_filter(
        report=monitoring_report,
        original_prompt=filtered_prompt,
        current_prompt=filtered_prompt,
    )

    final_image = pulsed_image
    retries_used = 0

    while needs_retry and retries_used < max_retries:
        print(f"اكتشفنا مشكلة – إعادة ترميم (محاولة {retries_used + 1}/{max_retries})...")
        
        # إعادة الـ pulse مع الـ correction_prompt
        pulsed_image = self.dna_inspired_color_pulse(
            img=repaired_geometry,
            mask=mask,
            prompt=correction_prompt,
            pulse_steps=pulse_steps,
            positive_boost=positive_boost,
            negative_suppress=negative_suppress,
            latent_mode=use_latent_pulse,
            pipeline=self.pipeline,
            latent_strength_base=latent_strength_base,
            latent_steps_per_pulse=latent_steps_per_pulse,
        )
        
        # إعادة Monitoring
        monitoring_report = self.monitor_repair_result(
            original=input_image,
            repaired=pulsed_image,
            mask=mask,
        )
        
        correction_prompt, needs_retry = self.post_monitor_filter(
            report=monitoring_report,
            original_prompt=filtered_prompt,
            current_prompt=correction_prompt,
        )
        
        retries_used += 1

    # ── 9. Final Polish (مع SSIM-aware تعديل) ──────────────────────────────
    orig_gray = cv2.cvtColor(np.array(input_image), cv2.COLOR_RGB2GRAY)
    final_gray = cv2.cvtColor(np.array(pulsed_image), cv2.COLOR_RGB2GRAY)
    ssim_score = ssim(orig_gray, final_gray, data_range=orig_gray.max() - orig_gray.min())

    if ssim_score > 0.93:
        sharpen_factor = final_sharpen * 1.15
        contrast_factor = final_contrast * 1.10
        color_factor = final_color * 1.08
    elif ssim_score > 0.85:
        sharpen_factor = final_sharpen
        contrast_factor = final_contrast
        color_factor = final_color
    else:
        sharpen_factor = final_sharpen * 0.80
        contrast_factor = final_contrast * 0.90
        color_factor = final_color * 0.95
        print("تحذير: SSIM منخفض – تم تقليل التلميع للحفاظ على الهيكل")

    final = ImageEnhance.Sharpness(pulsed_image).enhance(sharpen_factor)
    final = ImageEnhance.Contrast(final).enhance(contrast_factor)
    final = ImageEnhance.Color(final).enhance(color_factor)

    final.save(output_path)
    print(f"تم حفظ النتيجة النهائية → {output_path}")

    # تقرير نهائي شامل
    final_report = {
        "final_lpips": monitoring_report.get("lpips_score", 0.0),
        "final_ssim": ssim_score,
        "final_detail_loss": monitoring_report.get("detail_loss_ratio", 0.0),
        "final_quality_score": monitoring_report.get("overall_quality_score", 0.0),
        "retries_used": retries_used,
        "status": "success" if not needs_retry else "partial (max retries reached)",
        "final_prompt_used": correction_prompt,
        "issues_resolved": monitoring_report.get("issues", []),
        "applied_sharpen": round(sharpen_factor, 3),
        "applied_contrast": round(contrast_factor, 3),
        "applied_color": round(color_factor, 3),
    }

    return final, final_report
