def post_monitor_filter(
    report: Dict,
    original_prompt: str,
    current_prompt: str,
    max_iterations: int = 2,
) -> Tuple[str, bool]:
    """
    Post-Monitor Filter المحسن: يولّد prompt تصحيحي بناءً على تقرير Monitoring
    """
    if not report["needs_correction"]:
        return current_prompt, False  # النتيجة جيدة، ما فيش حاجة لتصحيح

    correction_additions = []

    # 1. مشاكل LPIPS مرتفع (جودة عامة منخفضة)
    if report.get("lpips_score", 0.0) > 0.30:
        correction_additions.append("sharper details, higher fidelity, realistic textures, 8k quality")

    # 2. فقدان تفاصيل (detail loss)
    if report.get("detail_loss_ratio", 0.0) > 0.08:
        correction_additions.append("enhance fine details, preserve original structure, high resolution, sharp focus")

    # 3. مشاكل ألوان (مثلاً saturation عالية أو منخفضة، color bleeding)
    if "ألوان غريبة" in report.get("issues", []) or report.get("color_deviation", 0.0) > 0.15:
        correction_additions.append("natural colors, balanced saturation, reduce color bleeding")

    # 4. تشوهات بنيوية (warp/deformation)
    if "تشوهات بنيوية" in report.get("issues", []):
        correction_additions.append("perfect anatomy, no deformation, symmetrical face, accurate proportions")

    # 5. إضافة عامة للتوازن
    if correction_additions:
        correction_additions.append("masterpiece, best quality, ultra detailed")

    # دمج التعديلات في prompt جديد
    new_prompt = current_prompt
    if correction_additions:
        new_prompt += ", " + ", ".join(correction_additions)

    return new_prompt, True  # يحتاج إعادة
